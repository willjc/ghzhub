	initSignaturePad() {
		// #ifdef H5
		let canvas = document.querySelector('#signatureCanvas')
		if (!canvas) {
			console.error('找不到 canvas 元素')
			uni.showToast({
				title: '初始化签名板失败',
				icon: 'none'
			})
			return
		}

		// uni-app H5 编译后可能包装了元素，需要获取真正的 canvas
		let realCanvas = canvas
		console.log('初始 canvas:', canvas, 'tagName:', canvas.tagName)
		
		// 如果获取到的不是真正的 canvas 元素，尝试查找内部的 canvas
		if (canvas.tagName !== 'CANVAS') {
			const innerCanvas = canvas.querySelector('canvas')
			if (innerCanvas) {
				realCanvas = innerCanvas
				console.log('找到内部 canvas:', innerCanvas)
			}
		}

		// 再次验证是否为真正的 canvas
		if (!realCanvas || realCanvas.tagName !== 'CANVAS' || typeof realCanvas.getContext !== 'function') {
			console.error('获取到的不是有效的 canvas 元素', realCanvas)
			uni.showToast({
				title: '签名板初始化失败',
				icon: 'none'
			})
			return
		}

		// 获取容器宽度（减去padding）
		const container = realCanvas.parentElement
		const containerWidth = container.offsetWidth
		this.canvasWidth = containerWidth > 60 ? containerWidth - 60 : 300

		// 强制更新canvas尺寸（等待Vue更新完成）
		this.$nextTick(() => {
			// 设置canvas的实际绘图尺寸
			realCanvas.width = this.canvasWidth
			realCanvas.height = this.canvasHeight

			console.log('Canvas尺寸:', this.canvasWidth, 'x', this.canvasHeight)
			console.log('Canvas元素类型:', realCanvas.tagName, 'getContext:', typeof realCanvas.getContext)

			// 初始化签名板
			try {
				this.signaturePad = new SmoothSignature(realCanvas, {
					width: this.canvasWidth,
					height: this.canvasHeight,
					minWidth: 2,
					maxWidth: 4,
					color: '#000000',
					bgColor: '#fafafa'
				})
				console.log('签名板初始化成功')
			} catch (error) {
				console.error('初始化签名板失败:', error)
				uni.showToast({
					title: '签名板初始化失败',
					icon: 'none'
				})
			}
		})
		// #endif
	},
